编写一个寻找连通子图的函数

from collections import defaultdict, deque

def get_sub_graphs(edges: list[list[int]]):
    graph = defaultdict(set)
    for u, v in edges:
        graph[u].add(v)
        graph[v].add(u)

    visited = set()
    graphs = []
    for node in graph:
        if node not in visited:
            sub_graph = []
            queue = deque([node])
            visited.add(node)
            while queue:
                cur = queue.popleft()
                sub_graph.append(cur)
                for neighbor in graph[cur]:
                    if neighbor not in visited:
                        visited.add(neighbor)
                        queue.append(neighbor)
            graphs.append(sub_graph)

    graphs.sort(key=len, reverse=True)
    return graphs

import time
import random

def generate_edges(node_count: int, avg_edges_per_node: float) -> list[list[int]]:
    edges = set()
    total_edges = int(node_count * avg_edges_per_node // 2)
    
    while len(edges) < total_edges:
        a = random.randint(0, node_count - 1)
        b = random.randint(0, node_count - 1)
        if a != b and (a, b) not in edges and (b, a) not in edges:
            edges.add((a, b))
            
    return list(edges)

改写为 Rust 版本 #

以下代码放在 my-rust-project/src/lib.rs 内

use pyo3::prelude::*;
use std::collections::{HashSet, HashMap, VecDeque};

#[pyfunction]
fn get_sub_graphs(edges: Vec<Vec<i32>>) -> Vec<Vec<i32>> {
    let mut graph: HashMap<i32, HashSet<i32>> = HashMap::new();
    for edge in edges {
        graph.entry(edge[0]).or_insert_with(HashSet::new).insert(edge[1]);
        graph.entry(edge[1]).or_insert_with(HashSet::new).insert(edge[0]);
    }

    let mut visited = HashSet::new();
    let mut graphs = Vec::new();

    for &node in graph.keys() {
        if !visited.contains(&node) {
            visited.insert(node);
            let mut queue = VecDeque::new();
            queue.push_back(node);
            let mut sub_graph = Vec::new();

            while let Some(cur) = queue.pop_front() {
                sub_graph.push(cur);
                for &neighbor in &graph[&cur] {
                    if visited.insert(neighbor) {
                        queue.push_back(neighbor);
                    }
                }
            }

            sub_graph.sort();
            graphs.push(sub_graph);
        }
    }

    graphs.sort_by(|a, b| b.len().cmp(&a.len()));
    graphs
}


/// A Python module implemented in Rust.
#[pymodule]
fn my-rust-project(_py: Python, m: &PyModule) -> PyResult<()> {
    m.add_function(wrap_pyfunction!(get_sub_graphs, m)?)?;
    Ok(())
}
